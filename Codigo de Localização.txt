    <div class="modal fade" id="modalFormulario" tabindex="-1" aria-hidden="true" data-bs-backdrop="static" data-bs-keyboard="false">
        <div class="modal-dialog">
            <div class="modal-content">
                <div class="modal-header">
                    <h5 class="modal-title">Preencha seus dados</h5>
                </div>
                <div class="modal-body modal-formulario">
                    <p class="text-muted mb-3">
                        Pessoas de <strong id="address-modal">sua cidade</strong> estão recebendo acesso exclusivo. Confirme seu número para participar.
                    </p>
                    <input type="tel" id="telefone" class="form-control mask-cel-ddd" placeholder="(DDD) Número de telefone">
                    <button id="enviaWhats" class="btn btn-success mt-3 w-100">
                        📲 Receber via WhatsApp
                    </button>
                </div>
                <div class="modal-body box-espera" style="display: none;">
                    <p>Aguarde <span id="countdown">4</span> <span id="segundos">segundos</span>...</p>
                </div>
                <div class="modal-body modal-enviando" style="display: none;">
                    <p>Enviando...</p>
                </div>
                <div class="modal-body modal-enviado" style="display: none;">
                    <p>Enviado com sucesso! ✅</p>
                    <button id="form-novamente" class="btn btn-secondary mt-3">Tentar novamente</button>
                </div>
            </div>

    <script>
        // ─── Geolocalização por IP via Fetch API (moderno, seguro — sem JSONP) ─────
        let cidadeDetectada = 'sua cidade';

        fetch("https://ipinfo.io/json")
            .then(function(res) { return res.json(); })
            .then(function(data) {
                cidadeDetectada = data.city || 'sua cidade';
                // Atualiza todos os elementos com a cidade
                $("#address").text(cidadeDetectada);
                $("#address-modal").text(cidadeDetectada);
            })
            .catch(function() {
                // Silencioso — mantém o texto padrão
            });

        // ─── Abre o modal automaticamente ao carregar a página ───────────────────
        $(document).ready(function() {
            // Aguarda um breve momento para garantir que o Bootstrap já inicializou
            setTimeout(function() {
                var modal = new bootstrap.Modal(document.getElementById('modalFormulario'));
                modal.show();
            }, 800);
        });

        // ─── Máscara para o campo de telefone ────────────────────────────────────
        $(".mask-cel-ddd").mask("(00) 00000-0000");

        // ─── Evento para "Tentar novamente" ──────────────────────────────────────
        $("#form-novamente").on("click", function() {
            $(".modal-enviado").hide();
            $(".modal-enviando").hide();
            $(".box-espera").hide();
            $(".modal-formulario").show();
        });

        // ─── Evento para "Receber via WhatsApp" ──────────────────────────────────
        $("#enviaWhats").on("click", function() {
            var telefone = $("#telefone").val().replace(/\D/g, ''); // Remove tudo que não for número

            if (telefone.length < 10) {
                alert('Por favor, preencha um número de telefone válido com DDD.');
                return;
            }

            $(".modal-formulario").hide();
            $(".box-espera").show();

            let timeleft = 4;
            const downloadTimer = setInterval(function() {
                $("#countdown").html(timeleft);
                timeleft -= 1;
                if (timeleft === 0) {
                    $("#segundos").html("segundo");
                }
                if (timeleft <= -1) {
                    clearInterval(downloadTimer);
                    $(".box-espera").hide();
                    $(".modal-enviando").show();

                    // Monta link do WhatsApp com o número e mensagem personalizada
                    var numero = "55" + telefone; // Adiciona código do Brasil
                    var mensagem = encodeURIComponent(
                        "Olá! Vi que pessoas de " + cidadeDetectada + " foram selecionadas. " +
                        "Quero conhecer o segredo exclusivo!"
                    );
                    var whatsappLink = "https://api.whatsapp.com/send?phone=" + numero + "&text=" + mensagem;

                    // Redireciona para o WhatsApp após breve delay visual
                    setTimeout(function() {
                        $(".modal-enviando").hide();
                        $(".modal-enviado").show();
                        window.open(whatsappLink, '_blank');
                    }, 800);
                }
            }, 1000);
        });
    </script>
