    <script>
        // Geolocalização com ipinfo.io
        $.get("https://ipinfo.io/json", function(response) {
            $("#address").html(response.city || "sua cidade");
        }, "jsonp");

        // Máscara para o campo de telefone
        $(".mask-cel-ddd").mask("(00) 00000-0000");

        // Evento para "Tentar novamente"
        $("#form-novamente").on("click", function() {
            $(".modal-enviado").hide();
            $(".modal-enviando").hide();
            $(".box-espera").hide();
            $(".modal-formulario").show();
        });

        // Evento para "Enviar via WhatsApp"
        $("#enviaWhats").on("click", function() {
            $(".modal-formulario").hide();
            $(".box-espera").show();
            $(".modal-enviando").show();

            let timeleft = 4;
            const downloadTimer = setInterval(function() {
                $("#countdown").html(timeleft);
                timeleft -= 1;
                if (timeleft === 0) {
                    $("#segundos").html("segundo");
                }
                if (timeleft <= -1) {
                    clearInterval(downloadTimer);
                    $(".modal-enviando").hide();
                    $(".modal-enviado").show();
                }
            }, 1000);
        });
    </script>
